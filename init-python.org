						  "#\\s-*In\\[.*\\]:"
						  "\\|"
						  "#\\s-*%%"

					     "\\|"
					     "#\\s-*%%"
					     "\\)\\s-*$") "")

	   (elpy-mode . flycheck-mode)

    :config

  (use-package blacken
    :straight t
    :hook (python-mode . blacken-mode)
    :config (setq blacken-line-length '80)
    :bind (:map python-mode-map ([remap indent-region] . blacken)))

  (setq elpy-modules (delq 'elpy-module-flymake elpy-modules))
